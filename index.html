📋 Google Sheets Setup Guide
Follow these steps once to connect your Google Spreadsheet as a live database. After setup, all Create / Read / Update / Delete operations happen in real-time with no refresh needed.
1
Create a Google Spreadsheet
Go to sheets.new to open a blank spreadsheet. Rename it EasyFlight DB. Create two sheets (tabs) at the bottom named exactly: Bookings and Clients (case-sensitive).
2
Add Headers — Bookings Tab (Row 1)
Click the Bookings tab. Paste this into cell A1 and press Enter — it will auto-fill all columns:
ID Timestamp From To DepartureDate ReturnDate FlightNumber Class Airline TripType FirstName LastName Email Phone Passport Nationality Adults Children Price Taxes Total Status Payment PaymentStatus Notes
⚠ Copy the headers by clicking the button in Settings → Google Sheets to avoid typos.
3
Add Headers — Clients Tab (Row 1)
Click the Clients tab and paste:
ID Timestamp FirstName LastName Email Phone DOB Nationality Passport PassportExpiry Address PreferredClass Status Meal FFNumber Notes
4
Open Apps Script Editor
In your spreadsheet click Extensions → Apps Script. A new tab opens with a code editor. Delete all existing code in Code.gs.
5
Paste the Apps Script Code
Copy the entire script below and paste it into the Apps Script editor. Then click Save (💾).
// ═══════════════════════════════════════════════════
// EasyFlight Enterprise — Google Apps Script Backend
// Paste entire file into Extensions > Apps Script > Code.gs
// Then deploy as Web App (Anyone access)
// ═══════════════════════════════════════════════════

const BOOKINGS_SHEET = 'Bookings';
const CLIENTS_SHEET  = 'Clients';

// CORS helper
function corsResponse(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}

// ─── GET handler (Read) ───────────────────────────
function doGet(e) {
  const action = e.parameter.action || '';
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  if (action === 'ping') {
    return corsResponse({ ok: true, msg: 'EasyFlight API online ✈' });
  }

  if (action === 'getBookings') {
    return corsResponse(readSheet(ss, BOOKINGS_SHEET));
  }

  if (action === 'getClients') {
    return corsResponse(readSheet(ss, CLIENTS_SHEET));
  }

  return corsResponse({ ok: false, error: 'Unknown action' });
}

function readSheet(ss, name) {
  const sheet = ss.getSheetByName(name);
  if (!sheet) return { ok: false, error: 'Sheet "' + name + '" not found' };
  const rows = sheet.getDataRange().getValues();
  if (rows.length < 2) return { ok: true, data: [] };
  const headers = rows[0].map(String);
  const data = rows.slice(1).map(row => {
    const obj = {};
    headers.forEach((h, i) => { obj[h] = row[i] !== undefined ? String(row[i]) : ''; });
    return obj;
  }).filter(r => r['ID'] && r['ID'].trim() !== '');
  return { ok: true, data };
}

// ─── POST handler (Create / Update / Delete) ──────
function doPost(e) {
  const lock = LockService.getScriptLock();
  lock.tryLock(10000);
  try {
    const body  = JSON.parse(e.postData.contents);
    const action = body.action || '';
    const ss    = SpreadsheetApp.getActiveSpreadsheet();

    // ── CREATE BOOKING ──
    if (action === 'addBooking') {
      const sheet = ss.getSheetByName(BOOKINGS_SHEET);
      const lastRow = sheet.getLastRow();
      const id = 'BK-' + String(lastRow).padStart(4, '0');
      const ts = new Date().toLocaleString('en-GB');
      const total = (parseFloat(body.Price||0) + parseFloat(body.Taxes||0)).toFixed(2);
      sheet.appendRow([
        id, ts,
        body.From, body.To, body.DepartureDate, body.ReturnDate,
        body.FlightNumber, body.Class, body.Airline, body.TripType,
        body.FirstName, body.LastName, body.Email, body.Phone,
        body.Passport, body.Nationality, body.Adults, body.Children,
        body.Price, body.Taxes, total,
        body.Status, body.Payment, body.PaymentStatus, body.Notes
      ]);
      return corsResponse({ ok: true, id, action: 'addBooking' });
    }

    // ── CREATE CLIENT ──
    if (action === 'addClient') {
      const sheet = ss.getSheetByName(CLIENTS_SHEET);
      const lastRow = sheet.getLastRow();
      const id = 'CL-' + String(lastRow).padStart(4, '0');
      const ts = new Date().toLocaleString('en-GB');
      sheet.appendRow([
        id, ts,
        body.FirstName, body.LastName, body.Email, body.Phone,
        body.DOB, body.Nationality, body.Passport, body.PassportExpiry,
        body.Address, body.PreferredClass, body.Status,
        body.Meal, body.FFNumber, body.Notes
      ]);
      return corsResponse({ ok: true, id, action: 'addClient' });
    }

    // ── UPDATE BOOKING ──
    if (action === 'updateBooking') {
      const sheet = ss.getSheetByName(BOOKINGS_SHEET);
      const rows  = sheet.getDataRange().getValues();
      const hdr   = rows[0].map(String);
      for (let i = 1; i < rows.length; i++) {
        if (String(rows[i][0]) === String(body.ID)) {
          // Update specified columns by name
          Object.keys(body.updates || {}).forEach(col => {
            const ci = hdr.indexOf(col);
            if (ci >= 0) sheet.getRange(i + 1, ci + 1).setValue(body.updates[col]);
          });
          return corsResponse({ ok: true, action: 'updateBooking', id: body.ID });
        }
      }
      return corsResponse({ ok: false, error: 'Booking not found: ' + body.ID });
    }

    // ── UPDATE CLIENT ──
    if (action === 'updateClient') {
      const sheet = ss.getSheetByName(CLIENTS_SHEET);
      const rows  = sheet.getDataRange().getValues();
      const hdr   = rows[0].map(String);
      for (let i = 1; i < rows.length; i++) {
        if (String(rows[i][0]) === String(body.ID)) {
          Object.keys(body.updates || {}).forEach(col => {
            const ci = hdr.indexOf(col);
            if (ci >= 0) sheet.getRange(i + 1, ci + 1).setValue(body.updates[col]);
          });
          return corsResponse({ ok: true, action: 'updateClient', id: body.ID });
        }
      }
      return corsResponse({ ok: false, error: 'Client not found: ' + body.ID });
    }

    // ── DELETE BOOKING ──
    if (action === 'deleteBooking') {
      const sheet = ss.getSheetByName(BOOKINGS_SHEET);
      const rows  = sheet.getDataRange().getValues();
      for (let i = 1; i < rows.length; i++) {
        if (String(rows[i][0]) === String(body.ID)) {
          sheet.deleteRow(i + 1);
          return corsResponse({ ok: true, action: 'deleteBooking', id: body.ID });
        }
      }
      return corsResponse({ ok: false, error: 'Not found' });
    }

    // ── DELETE CLIENT ──
    if (action === 'deleteClient') {
      const sheet = ss.getSheetByName(CLIENTS_SHEET);
      const rows  = sheet.getDataRange().getValues();
      for (let i = 1; i < rows.length; i++) {
        if (String(rows[i][0]) === String(body.ID)) {
          sheet.deleteRow(i + 1);
          return corsResponse({ ok: true, action: 'deleteClient', id: body.ID });
        }
      }
      return corsResponse({ ok: false, error: 'Not found' });
    }

    return corsResponse({ ok: false, error: 'Unknown action: ' + action });

  } catch (err) {
    return corsResponse({ ok: false, error: err.toString() });
  } finally {
    lock.releaseLock();
  }
}
6
Deploy as Web App
In the Apps Script editor:
1. Click Deploy → New deployment
2. Click the ⚙ gear icon → select Web app
3. Description: EasyFlight API
4. Execute as: Me
5. Who has access: Anyone
6. Click Deploy → authorize when prompted → copy the Web app URL
7
Connect in Settings
Go to Settings → Google Sheets, paste your Web App URL, and click Save & Test. You'll see a green "Connected" indicator when it works.
8
Deploy Your Site (Optional)
This file is a single HTML file — no server needed.
Free hosting options:
• GitHub Pages — push file to repo, enable Pages
• Netlify — drag-and-drop the HTML file
• Vercel — connect GitHub repo
• Any static hosting (Cloudflare Pages, Firebase Hosting)

The Google Sheets backend works from any domain because Apps Script handles CORS.
